## 2. 技术方案

### 2.1 轻量级架构 (推荐个人使用)

```python
import sqlite3
import json
from datetime import datetime
from typing import Dict, List, Optional

class PersonalAssistantMemory:
    """个人助理记忆系统 (轻量级)"""
    
    def __init__(self, db_path: str = "~/.personal_assistant.db"):
        self.db_path = db_path
        self._init_db()
    
    def _init_db(self):
        """初始化数据库"""
        conn = sqlite3.connect(self.db_path)
        cursor = conn.cursor()
        
        # 用户画像表
        cursor.execute("""
            CREATE TABLE IF NOT EXISTS user_profile (
                key TEXT PRIMARY KEY,
                value TEXT,
                category TEXT,
                updated_at TIMESTAMP
            )
        """)
        
        # 偏好表
        cursor.execute("""
            CREATE TABLE IF NOT EXISTS preferences (
                id INTEGER PRIMARY KEY AUTOINCREMENT,
                category TEXT,
                item TEXT,
                value TEXT,
                confidence REAL DEFAULT 0.5,
                created_at TIMESTAMP
            )
        """)
        
        # 日程表
        cursor.execute("""
            CREATE TABLE IF NOT EXISTS schedule (
                id INTEGER PRIMARY KEY AUTOINCREMENT,
                title TEXT,
                start_time TIMESTAMP,
                end_time TIMESTAMP,
                location TEXT,
                reminder BOOLEAN DEFAULT TRUE,
                completed BOOLEAN DEFAULT FALSE
            )
        """)
        
        # 对话历史表 (向量检索用，这里简化为全文检索)
        cursor.execute("""
            CREATE TABLE IF NOT EXISTS conversations (
                id INTEGER PRIMARY KEY AUTOINCREMENT,
                role TEXT,
                content TEXT,
                created_at TIMESTAMP
            )
        """)
        
        conn.commit()
        conn.close()
    
    # ========== 用户画像管理 ==========
    
    def set_profile(self, key: str, value: str, category: str = "basic"):
        """设置用户画像"""
        conn = sqlite3.connect(self.db_path)
        cursor = conn.cursor()
        cursor.execute("""
            INSERT OR REPLACE INTO user_profile (key, value, category, updated_at)
            VALUES (?, ?, ?, ?)
        """, (key, value, category, datetime.now()))
        conn.commit()
        conn.close()
    
    def get_profile(self, key: str = None) -> Dict:
        """获取用户画像"""
        conn = sqlite3.connect(self.db_path)
        cursor = conn.cursor()
        
        if key:
            cursor.execute("SELECT key, value FROM user_profile WHERE key = ?", (key,))
            result = dict(cursor.fetchone()) if cursor.fetchone() else {}
        else:
            cursor.execute("SELECT key, value, category FROM user_profile")
            result = {
                "basic": {},
                "work": {},
                "personal": {}
            }
            for row in cursor.fetchall():
                result[row[2]][row[0]] = row[1]
        
        conn.close()
        return result
    
    # ========== 偏好管理 ==========
    
    def add_preference(self, category: str, item: str, value: str, confidence: float = 0.8):
        """添加偏好"""
        conn = sqlite3.connect(self.db_path)
        cursor = conn.cursor()
        cursor.execute("""
            INSERT INTO preferences (category, item, value, confidence, created_at)
            VALUES (?, ?, ?, ?, ?)
        """, (category, item, value, confidence, datetime.now()))
        conn.commit()
        conn.close()
    
    def get_preferences(self, category: str = None) -> Dict:
        """获取偏好"""
        conn = sqlite3.connect(self.db_path)
        cursor = conn.cursor()
        
        if category:
            cursor.execute("""
                SELECT item, value, confidence FROM preferences 
                WHERE category = ? 
                ORDER BY confidence DESC
            """, (category,))
        else:
            cursor.execute("""
                SELECT category, item, value, confidence FROM preferences 
                ORDER BY category, confidence DESC
            """)
        
        result = {}
        for row in cursor.fetchall():
            if category:
                result[row[0]] = {"value": row[1], "confidence": row[2]}
            else:
                if row[0] not in result:
                    result[row[0]] = []
                result[row[0]].append({"item": row[1], "value": row[2], "confidence": row[3]})
        
        conn.close()
        return result
    
    # ========== 日程管理 ==========
    
    def add_schedule(self, title: str, start_time: datetime, end_time: datetime,
                    location: str = None, reminder: bool = True):
        """添加日程"""
        conn = sqlite3.connect(self.db_path)
        cursor = conn.cursor()
        cursor.execute("""
            INSERT INTO schedule (title, start_time, end_time, location, reminder)
            VALUES (?, ?, ?, ?, ?)
        """, (title, start_time, end_time, location, reminder))
        conn.commit()
        conn.close()
    
    def get_today_schedule(self) -> List:
        """获取今日日程"""
        conn = sqlite3.connect(self.db_path)
        cursor = conn.cursor()
        
        today = datetime.now().date()
        cursor.execute("""
            SELECT title, start_time, end_time, location, reminder, completed
            FROM schedule 
            WHERE date(start_time) = ? AND completed = FALSE
            ORDER BY start_time
        """, (today,))
        
        result = [
            {
                "title": row[0],
                "start_time": row[1],
                "end_time": row[2],
                "location": row[3],
                "reminder": row[4],
                "completed": row[5]
            }
            for row in cursor.fetchall()
        ]
        
        conn.close()
        return result
    
    def get_upcoming_schedule(self, hours: int = 24) -> List:
        """获取未来日程"""
        conn = sqlite3.connect(self.db_path)
        cursor = conn.cursor()
        
        from datetime import timedelta
        now = datetime.now()
        future = now + timedelta(hours=hours)
        
        cursor.execute("""
            SELECT title, start_time, end_time, location
            FROM schedule 
            WHERE start_time BETWEEN ? AND ? AND completed = FALSE
            ORDER BY start_time
        """, (now, future))
        
        result = [
            {
                "title": row[0],
                "start_time": row[1],
                "end_time": row[2],
                "location": row[3]
            }
            for row in cursor.fetchall()
        ]
        
        conn.close()
        return result
    
    # ========== 对话历史 ==========
    
    def add_conversation(self, role: str, content: str):
        """添加对话"""
        conn = sqlite3.connect(self.db_path)
        cursor = conn.cursor()
        cursor.execute("""
            INSERT INTO conversations (role, content, created_at)
            VALUES (?, ?, ?)
        """, (role, content, datetime.now()))
        conn.commit()
        conn.close()
    
    def get_recent_conversations(self, limit: int = 20) -> List:
        """获取最近对话"""
        conn = sqlite3.connect(self.db_path)
        cursor = conn.cursor()
        cursor.execute("""
            SELECT role, content, created_at FROM conversations 
            ORDER BY created_at DESC 
            LIMIT ?
        """, (limit,))
        
        result = [
            {"role": row[0], "content": row[1], "created_at": row[2]}
            for row in cursor.fetchall()
        ]
        
        conn.close()
        return list(reversed(result))
    
    # ========== 记忆检索 ==========
    
    def search_memories(self, query: str) -> List:
        """搜索记忆 (简化版全文检索)"""
        conn = sqlite3.connect(self.db_path)
        cursor = conn.cursor()
        
        # 在多个表中搜索
        results = []
        
        # 搜索用户画像
        cursor.execute("""
            SELECT 'profile' as type, key, value FROM user_profile 
            WHERE value LIKE ?
        """, (f"%{query}%",))
        results.extend([{"type": "profile", "key": row[1], "value": row[2]} for row in cursor.fetchall()])
        
        # 搜索偏好
        cursor.execute("""
            SELECT 'preference' as type, category, item, value FROM preferences 
            WHERE value LIKE ? OR item LIKE ?
        """, (f"%{query}%", f"%{query}%"))
        results.extend([{"type": "preference", "category": row[1], "item": row[2], "value": row[3]} for row in cursor.fetchall()])
        
        # 搜索对话
        cursor.execute("""
            SELECT 'conversation' as type, role, content FROM conversations 
            WHERE content LIKE ?
            ORDER BY created_at DESC
            LIMIT 10
        """, (f"%{query}%",))
        results.extend([{"type": "conversation", "role": row[1], "content": row[2]} for row in cursor.fetchall()])
        
        conn.close()
        return results
    
    # ========== 导出/删除 (隐私) ==========
    
    def export_all_data(self) -> Dict:
        """导出所有数据 (GDPR)"""
        return {
            "profile": self.get_profile(),
            "preferences": self.get_preferences(),
            "schedule": self._export_schedule(),
            "conversations": self._export_conversations()
        }
    
    def _export_schedule(self) -> List:
        conn = sqlite3.connect(self.db_path)
        cursor = conn.cursor()
        cursor.execute("SELECT * FROM schedule")
        result = cursor.fetchall()
        conn.close()
        return result
    
    def _export_conversations(self) -> List:
        conn = sqlite3.connect(self.db_path)
        cursor = conn.cursor()
        cursor.execute("SELECT * FROM conversations")
        result = cursor.fetchall()
        conn.close()
        return result
    
    def delete_all_data(self):
        """删除所有数据 (被遗忘权)"""
        conn = sqlite3.connect(self.db_path)
        cursor = conn.cursor()
        
        cursor.execute("DELETE FROM user_profile")
        cursor.execute("DELETE FROM preferences")
        cursor.execute("DELETE FROM schedule")
        cursor.execute("DELETE FROM conversations")
        
        conn.commit()
        conn.close()

# 使用示例
if __name__ == "__main__":
    memory = PersonalAssistantMemory()
    
    # 设置用户画像
    memory.set_profile("name", "张三", "basic")
    memory.set_profile("occupation", "软件工程师", "work")
    memory.set_profile("location", "北京", "basic")
    
    # 添加偏好
    memory.add_preference("food", "coffee", "冰美式", 0.9)
    memory.add_preference("food", "spicy", "少辣", 0.7)
    
    # 添加日程
    from datetime import datetime, timedelta
    tomorrow = datetime.now() + timedelta(days=1)
    memory.add_schedule(
        "产品评审会",
        tomorrow.replace(hour=10, minute=0),
        tomorrow.replace(hour=11, minute=0),
        "会议室 A"
    )
    
    # 获取信息
    print("用户画像:", memory.get_profile())
    print("偏好:", memory.get_preferences())
    print("今日日程:", memory.get_today_schedule())
    print("未来日程:", memory.get_upcoming_schedule(48))
    
    # 搜索记忆
    results = memory.search_memories("咖啡")
    print("搜索结果:", results)
```
