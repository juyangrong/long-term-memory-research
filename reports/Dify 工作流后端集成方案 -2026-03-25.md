# Dify 工作流后端集成方案

**报告日期**: 2026-03-25  
**目标**: 将 Dify 工作流封装为本地方法，供后端服务透明调用

---

## 📋 执行摘要

### 核心需求
将 Dify 工作流 API 封装成**本地方法调用**，使后端服务无需关心 HTTP 细节，像调用普通服务方法一样使用 Dify 工作流。

### 推荐方案
**适配器模式 + 仓库模式**，通过 SDK 封装 HTTP 调用，提供类型安全的本地方法接口。

---

### 1.1 整体架构

```

│                        后端服务层                                │
├─────────────────────────────────────────────────────────────────┤
│  业务代码                                                        │
│  ↓                                                              │
│  DifyWorkflowService (本地方法接口)                              │
│  ↓                                                              │
├─────────────────────────────────────────────────────────────────┤
│  适配器层 (Dify Adapter)                                         │
│  - DifyWorkflowAdapter                                          │
│  - 参数转换/验证                                                 │
│  - 响应解析/映射                                                 │
├─────────────────────────────────────────────────────────────────┤
│  SDK 层 (Dify SDK)                                               │
│  - HTTP 客户端封装                                                │
│  - 认证管理                                                      │
│  - 错误处理                                                      │
├─────────────────────────────────────────────────────────────────┤
│  网络层                                                          │
│  - HTTPS 请求                                                    │
│  - SSE 流式解析                                                   │
└─────────────────────────────────────────────────────────────────┘
                              ↓
                    Dify Cloud/API
```

### 1.2 设计模式

| 模式 | 作用 | 实现位置 |
|------|------|----------|
| **适配器模式** | 将 Dify API 转换为本地方法签名 | DifyWorkflowAdapter |
| **仓库模式** | 抽象工作流调用，隔离数据源 | DifyWorkflowRepository |
| **工厂模式** | 创建不同类型的工作流客户端 | DifyClientFactory |
| **策略模式** | 支持同步/流式不同响应模式 | ResponseModeStrategy |

```python
# dify_workflow_adapter.py
import os
import json
import requests
from typing import Any, Dict, Optional, Generator, Union
from dataclasses import dataclass
from enum import Enum

class ResponseMode(Enum):
    BLOCKING = "blocking"
    STREAMING = "streaming"

@dataclass
class WorkflowResult:
    """工作流执行结果"""
    success: bool
    data: Optional[Dict[str, Any]] = None
    error: Optional[str] = None
    conversation_id: Optional[str] = None
    message_id: Optional[str] = None

class DifyWorkflowAdapter:
    """Dify 工作流适配器 - 将 API 封装为本地方法"""
    
    def __init__(
        self,
        api_key: Optional[str] = None,
        base_url: str = "https://api.dify.ai/v1",
        workflow_id: Optional[str] = None,
        timeout: tuple = (10, 90)
    ):
        self.api_key = api_key or os.getenv("DIFY_API_KEY")
        self.base_url = base_url
        self.workflow_id = workflow_id or os.getenv("DIFY_WORKFLOW_ID")
        self.timeout = timeout
        self._session = requests.Session()
        self._session.headers.update({
            "Authorization": f"Bearer {self.api_key}",
            "Content-Type": "application/json"
        })
    
    def execute(
        self,
        inputs: Dict[str, Any],
        user: Optional[str] = None,
        conversation_id: Optional[str] = None,
        response_mode: ResponseMode = ResponseMode.BLOCKING
    ) -> Union[WorkflowResult, Generator[Dict, None, None]]:
        """
        执行工作流 - 核心方法
        
        Args:
            inputs: 工作流输入参数
            user: 用户标识 (用于上下文隔离)
            conversation_id: 会话 ID (用于多轮对话)
            response_mode: 响应模式 (blocking/streaming)
        
        Returns:
            blocking 模式：WorkflowResult
            streaming 模式：Generator[Dict, None, None]
        """
        url = f"{self.base_url}/workflows/{self.workflow_id}/run"
        
        payload = {
            "inputs": inputs,
            "response_mode": response_mode.value,
            "user": user or self._generate_user_id()
        }
        
        if conversation_id:
            payload["conversation_id"] = conversation_id
        
        if response_mode == ResponseMode.BLOCKING:
            return self._execute_blocking(url, payload)
        else:
            return self._execute_streaming(url, payload)
    
    def _execute_blocking(self, url: str, payload: dict) -> WorkflowResult:
        """同步阻塞执行"""
        try:
            response = self._session.post(
                url,
                json=payload,
                timeout=self.timeout
            )
            response.raise_for_status()
            result = response.json()
            
            return WorkflowResult(
                success=True,
                data=result.get("data", {}),
                conversation_id=result.get("conversation_id"),
                message_id=result.get("message_id")
            )
        except requests.exceptions.Timeout:
            return WorkflowResult(
                success=False,
                error="Request timed out"
            )
        except requests.exceptions.HTTPError as e:
            return WorkflowResult(
                success=False,
                error=f"HTTP error: {e.response.status_code} - {e.response.text}"
            )
        except Exception as e:
            return WorkflowResult(
                success=False,
                error=f"Unexpected error: {str(e)}"
            )
    
    def _execute_streaming(self, url: str, payload: dict) -> Generator[Dict, None, None]:
        """流式执行 (SSE)"""
        try:
            response = self._session.post(
                url,
                json=payload,
                timeout=self.timeout,
                stream=True
            )
            response.raise_for_status()
            
            for line in response.iter_lines():
                if line:
                    line = line.decode('utf-8')
                    if line.startswith('data:'):
                        data = json.loads(line[5:].strip())
                        yield data
                    elif line == 'done:':
                        break
        except Exception as e:
            yield {"error": str(e), "event": "error"}
    
    def _generate_user_id(self) -> str:
        """生成用户标识"""
        import uuid
        return f"user_{uuid.uuid4().hex[:16]}"
    
    async def upload_file(self, file_path: str) -> Optional[str]:
        """
        上传文件获取 file_id (用于多模态输入)
        
        Returns:
            file_id 或 None
        """
        url = f"{self.base_url}/files/upload"
        try:
            with open(file_path, 'rb') as f:
                files = {'file': f}
                headers = {"Authorization": f"Bearer {self.api_key}"}
                response = requests.post(url, headers=headers, files=files)
                response.raise_for_status()
                result = response.json()
                return result.get("id")
        except Exception as e:
            print(f"File upload failed: {e}")
            return None
```

```python
# 1. 基础用法 - 同步调用
from dify_workflow_adapter import DifyWorkflowAdapter, ResponseMode

# 初始化适配器
dify = DifyWorkflowAdapter(
    api_key="sk-xxx",
    workflow_id="workflow_xxx"
)

# 像本地方法一样调用

    inputs={"query": "今天天气怎么样", "user_id": "12345"},
    user="user_abc"
)

if result.success:
    print(f"结果：{result.data}")
    print(f"会话 ID: {result.conversation_id}")
else:
    print(f"错误：{result.error}")

# 2. 流式调用
for chunk in dify.execute(
    inputs={"query": "写一篇长文章"},
    response_mode=ResponseMode.STREAMING
):
    if "answer" in chunk:
        print(chunk["answer"], end="", flush=True)

# 3. 带文件上传的多模态调用
file_id = await dify.upload_file("/path/to/document.pdf")
result = dify.execute(
    inputs={
        "document": {
            "type": "document",
            "transfer_method": "local_file",
            "file_id": file_id
        },
        "instruction": "总结这份文档"
    }
)
```

#### 核心实现

```java
// DifyWorkflowService.java
package com.example.service;

import io.github.guoshiqiufeng.dify.workflow.client.DifyWorkflowClient;
import io.github.guoshiqiufeng.dify.workflow.dto.request.WorkflowRunRequest;
import io.github.guoshiqiufeng.dify.workflow.dto.response.WorkflowRunResponse;
import lombok.RequiredArgsConstructor;
import lombok.extern.slf4j.Slf4j;
import org.springframework.stereotype.Service;
import reactor.core.publisher.Flux;

import java.util.Map;

@Service
@RequiredArgsConstructor
@Slf4j
public class DifyWorkflowService {
    
    private final DifyWorkflowClient workflowClient;
    
    /**
     * 同步执行工作流 - 像本地方法一样调用
     */
    public WorkflowResult execute(Map<String, Object> inputs, String userId) {
        try {
            WorkflowRunRequest request = WorkflowRunRequest.builder()
                .inputs(inputs)
                .user(userId)
                .responseMode("blocking")
                .build();
            
            WorkflowRunResponse response = workflowClient.run(request);
            
            return WorkflowResult.builder()
                .success(true)
                .data(response.getData())
                .conversationId(response.getConversationId())
                .messageId(response.getMessageId())
                .build();
                
        } catch (Exception e) {
            log.error("Dify workflow execution failed", e);
            return WorkflowResult.builder()
                .success(false)
                .error(e.getMessage())
                .build();
        }
    }
    
    /**
     * 流式执行工作流
     */
    public Flux<Map<String, Object>> executeStreaming(
        Map<String, Object> inputs, 
        String userId
    ) {
        WorkflowRunRequest request = WorkflowRunRequest.builder()
            .inputs(inputs)
            .user(userId)
            .responseMode("streaming")
            .build();
        
        return workflowClient.runStream(request)
            .doOnError(e -> log.error("Stream error", e));
    }
}

// WorkflowResult.java
@Data
@Builder
public class WorkflowResult {
    private boolean success;
    private Map<String, Object> data;
    private String error;
    private String conversationId;
    private String messageId;
}
```

#### 使用示例

```java
// Controller 调用
@RestController
@RequestMapping("/api/ai")
@RequiredArgsConstructor
public class AIController {
    
    private final DifyWorkflowService difyService;
    
    @PostMapping("/query")
    public ResponseEntity<WorkflowResult> query(@RequestBody QueryRequest request) {
        // 像调用本地方法一样使用 Dify 工作流
        WorkflowResult result = difyService.execute(
            Map.of("query", request.getQuery(), "user_id", request.getUserId()),
            request.getUserId()
        );
        
        if (result.isSuccess()) {
            return ResponseEntity.ok(result);
        } else {
            return ResponseEntity
                .status(HttpStatus.INTERNAL_SERVER_ERROR)
                .body(result);
        }
    }
    
    @PostMapping(value = "/stream", produces = MediaType.TEXT_EVENT_STREAM_VALUE)
    public Flux<Map<String, Object>> streamQuery(@RequestBody QueryRequest request) {
        // 流式返回 SSE
        return difyService.executeStreaming(
            Map.of("query", request.getQuery()),
            request.getUserId()
        );
    }
}
```

```python
from functools import lru_cache
import hashlib

@lru_cache(maxsize=1000)
def cached_execute(query_hash: str, inputs_json: str):
    """缓存相同查询的结果"""
    pass

def execute_with_cache(self, inputs):
    query_hash = hashlib.md5(
        json.dumps(inputs, sort_keys=True).encode()
    ).hexdigest()
    return cached_execute(query_hash, json.dumps(inputs))
```

## 🐛 七、错误处理

### 7.1 常见错误码

| 状态码 | 原因 | 解决方案 |
|--------|------|----------|
| 400 | inputs 结构错误 | 用 JSON Schema 验证 |
| 401 | API 密钥无效 | 轮换密钥，告警 |
| 403 | 频率超限 | 实现令牌桶限流 |
| 422 | 文件类型不匹配 | 工作流健康检查 |
| 500 | Dify 服务端错误 | 重试 + 降级 |

### 7.2 错误处理模板

```python
class DifyError(Exception):
    """Dify 自定义异常"""
    pass

class DifyTimeoutError(DifyError):
    pass

class DifyAuthError(DifyError):
    pass

def execute_safe(self, inputs, user):
    try:
        result = self.execute(inputs, user)
        if not result.success:
            raise DifyError(result.error)
        return result
    except DifyTimeoutError:
        # 降级逻辑
        return fallback_result()
    except DifyAuthError:
        # 密钥轮换
        rotate_api_key()
        raise
```
