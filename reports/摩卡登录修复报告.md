  const isChecked = await checkbox.isChecked();
  if (!isChecked) {
    await checkbox.click();
    console.log('✅ 隐私政策已勾选');
    
    // 验证是否勾选成功
    const isNowChecked = await checkbox.isChecked();
    if (isNowChecked) {
      console.log('✅ 验证成功：复选框已勾选');
    }
  }
}
