## 1. 技术概述

### 1.1 核心原理

生物启发式记忆模仿人脑的记忆机制，包括情景记忆、语义记忆、程序性记忆的分工，以及记忆巩固、遗忘曲线等生物特性。

```
人脑记忆系统:
  情景记忆 (海马体) → 具体事件、时间、地点
  语义记忆 (新皮层) → 抽象知识、概念
  程序性记忆 (基底节) → 技能、习惯

AI 类比:
  情景记忆 → 对话历史、事件记录
  语义记忆 → 知识库、事实
  程序性记忆 → 技能、工作流
```

### 1.2 关键机制

| 机制 | 生物原理 | AI 实现 |
|------|----------|---------|
| **记忆巩固** | 睡眠中重放记忆 | 离线记忆整理 |
| **遗忘曲线** | 艾宾浩斯遗忘 | 时间衰减权重 |
| **情感标记** | 杏仁核调节 | 情感分数加权 |
| **模式分离** | 区分相似记忆 | 对比学习 |
| **模式完成** | 部分线索回忆 | 补全检索 |

```python
def sleep_replay(memory_system, replay_ratio=0.1):
    """离线时重放重要记忆"""
    # 选择重要记忆 (高情感分数、高访问频率)
    important_memories = select_important_memories(memory_system)
    
    # 重放并强化
    for memory in important_memories[:int(len(important_memories) * replay_ratio)]:
```

2. **艾宾浩斯遗忘曲线**
```python
def forgetting_curve(initial_strength, time_days):
    """艾宾浩斯遗忘曲线"""
    # R = e^(-t/S)
    import math
    return initial_strength * math.exp(-time_days / 7)  # 7 天半衰期

def apply_forgetting(memory_system):
    """应用遗忘"""
    current_time = datetime.now()
    for memory in memory_system.all_memories:
        age_days = (current_time - memory.created_at).days
        memory.strength = forgetting_curve(memory.initial_strength, age_days)
    
    # 删除强度低于阈值的记忆
    memory_system.delete_below_threshold(0.2)
```

3. **情感标记**
```python
def emotional_weighting(content: str, emotion_score: float):
    """情感加权"""
    # 高情感强度记忆优先巩固
    base_importance = calculate_base_importance(content)
    emotional_importance = base_importance * (1 + emotion_score)
```

## 3. 实施方案

### 3.1 简化版生物启发式记忆

```python
import sqlite3
import math
from datetime import datetime, timedelta
from typing import Dict, List

class BioInspiredMemory:
    def __init__(self, db_path: str = "~/.bio_memory.db"):
        self.db_path = db_path
        self._init_db()
        
        # 遗忘曲线参数
        self.half_life_days = 7  # 半衰期
        self.consolidation_threshold = 0.5
        
    def _init_db(self):
        conn = sqlite3.connect(self.db_path)
        cursor = conn.cursor()
        
        cursor.execute("""
            CREATE TABLE IF NOT EXISTS memories (
                id INTEGER PRIMARY KEY AUTOINCREMENT,
                user_id TEXT,
                content TEXT,
                memory_type TEXT,  -- episodic/semantic/procedural
                emotional_score REAL DEFAULT 0.5,
                strength REAL DEFAULT 1.0,
                initial_strength REAL DEFAULT 1.0,
                created_at TIMESTAMP,
                last_accessed TIMESTAMP,
                access_count INTEGER DEFAULT 0,
                consolidated BOOLEAN DEFAULT FALSE
            )
        """)
        
        conn.commit()
        conn.close()
    
    def add_memory(self, user_id: str, content: str, 
                   memory_type: str = "episodic",
                   emotional_score: float = 0.5):
        """添加记忆"""
        conn = sqlite3.connect(self.db_path)
        cursor = conn.cursor()
        
        cursor.execute("""
            INSERT INTO memories (user_id, content, memory_type, 
                                  emotional_score, created_at)
            VALUES (?, ?, ?, ?, ?)
        """, (user_id, content, memory_type, emotional_score, datetime.now()))
        
        conn.commit()
        conn.close()
    
    def forgetting_curve(self, initial_strength: float, time_days: float) -> float:
        """艾宾浩斯遗忘曲线"""
        return initial_strength * math.exp(-time_days / self.half_life_days)
    
    def calculate_strength(self, memory: Dict) -> float:
        """计算当前记忆强度"""
        age_days = (datetime.now() - memory["created_at"]).days
        access_boost = math.log1p(memory["access_count"]) * 0.1  # 访问次数提升
        
        base_strength = self.forgetting_curve(
            memory["initial_strength"], 
            age_days
        )
        
        # 情感加权
        emotional_boost = memory["emotional_score"] * 0.2
        
        # 巩固加成
        consolidation_boost = 0.1 if memory["consolidated"] else 0
        
        return min(base_strength + access_boost + emotional_boost + consolidation_boost, 1.0)
    
    def retrieve(self, user_id: str, query: str, limit: int = 10) -> List:
        """检索记忆 (考虑强度)"""
        conn = sqlite3.connect(self.db_path)
        cursor = conn.cursor()
        
        # 简单实现：按强度排序
        cursor.execute("""
            SELECT * FROM memories 
            WHERE user_id = ? AND strength > 0.3
            ORDER BY strength DESC, last_accessed DESC
            LIMIT ?
        """, (user_id, limit))
        
        results = []
        for row in cursor.fetchall():
            memory = {
                "id": row[0],
                "user_id": row[1],
                "content": row[2],
                "memory_type": row[3],
                "emotional_score": row[4],
                "strength": row[5],
                "initial_strength": row[6],
                "created_at": row[7],
                "last_accessed": row[8],
                "access_count": row[9],
                "consolidated": row[10]
            }
            memory["current_strength"] = self.calculate_strength(memory)
            results.append(memory)
        
        conn.close()
        return results
    
    def access_memory(self, memory_id: int):
        """访问记忆 (更新访问记录)"""
        conn = sqlite3.connect(self.db_path)
        cursor = conn.cursor()
        
        cursor.execute("""
            UPDATE memories 
            SET access_count = access_count + 1,
                last_accessed = ?
            WHERE id = ?
        """, (datetime.now(), memory_id))
        
        conn.commit()
        conn.close()
    
    def consolidate(self, user_id: str):
        """记忆巩固 (离线任务)"""
        conn = sqlite3.connect(self.db_path)
        cursor = conn.cursor()
        
        # 选择高强度但未巩固的记忆
        cursor.execute("""
            SELECT id FROM memories 
            WHERE user_id = ? AND consolidated = FALSE AND strength > ?
            ORDER BY strength DESC
            LIMIT 10
        """, (user_id, self.consolidation_threshold))
        
        to_consolidate = [row[0] for row in cursor.fetchall()]
        
        # 标记为已巩固
        for memory_id in to_consolidate:
            cursor.execute("""
                UPDATE memories 
                SET consolidated = TRUE
                WHERE id = ?
            """, (memory_id,))
        
        conn.commit()
        conn.close()
        
        return len(to_consolidate)
    
    def cleanup(self, user_id: str):
        """清理低强度记忆"""
        conn = sqlite3.connect(self.db_path)
        cursor = conn.cursor()
        
        # 删除强度低于阈值的记忆
        cursor.execute("""
            DELETE FROM memories 
            WHERE user_id = ? AND strength < 0.1 AND access_count < 2
        """, (user_id,))
        
        deleted = cursor.rowcount
        conn.commit()
        conn.close()
        
        return deleted

# 使用示例
if __name__ == "__main__":
    memory = BioInspiredMemory()
    
    # 添加记忆 (不同情感强度)
    memory.add_memory("user_123", "第一次见到张三", "episodic", emotional_score=0.9)
    memory.add_memory("user_123", "喜欢喝咖啡", "semantic", emotional_score=0.6)
    memory.add_memory("user_123", "每天写代码", "procedural", emotional_score=0.3)
    
    # 检索
    results = memory.retrieve("user_123", "")
    for r in results:
        print(f"记忆：{r['content']}, 强度：{r['current_strength']:.2f}")
    
    # 访问记忆 (增强强度)
    memory.access_memory(1)
    
    # 定期巩固 (每天运行)
    consolidated = memory.consolidate("user_123")
    print(f"巩固了 {consolidated} 条记忆")
    
    # 定期清理 (每周运行)
    deleted = memory.cleanup("user_123")
    print(f"清理了 {deleted} 条记忆")
```

```python
def classify_memory_type(content: str) -> str:
    """分类记忆类型"""
    # 简化实现，实际应该用分类模型
    if any(kw in content for kw in ["今天", "昨天", "见面", "经历"]):

    elif any(kw in content for kw in ["喜欢", "偏好", "习惯", "认为"]):
        return "semantic"

        return "procedural"
    else:
        return "semantic"
```

### 4.2 情感评分

```python
def estimate_emotional_score(content: str) -> float:
    """估计情感分数 (简化版)"""
    positive_words = ["喜欢", "爱", "开心", "满意", "棒", "好"]
    negative_words = ["讨厌", "恨", "生气", "失望", "差", "坏"]
    neutral_words = ["是", "有", "在", "了"]
    
    pos_count = sum(1 for w in positive_words if w in content)
    neg_count = sum(1 for w in negative_words if w in content)
    
    # 情感强度 = |正 - 负| / 总词数
    total = pos_count + neg_count
    if total == 0:
        return 0.5  # 中性
    
    intensity = abs(pos_count - neg_count) / total
    return min(0.5 + intensity, 1.0)
```

### 4.3 离线巩固任务

```python
import schedule
import time

def daily_consolidation():
    """每日巩固任务"""
    memory = BioInspiredMemory()
    
    # 获取所有用户
    users = get_all_users()
    
    for user_id in users:
        # 巩固
        consolidated = memory.consolidate(user_id)
        log(f"用户 {user_id} 巩固了 {consolidated} 条记忆")
        
        # 清理 (每周)
        if datetime.now().weekday() == 6:  # 周日
            deleted = memory.cleanup(user_id)
            log(f"用户 {user_id} 清理了 {deleted} 条记忆")

# 每天凌晨 3 点运行
schedule.every().day.at("03:00").do(daily_consolidation)

while True:
    schedule.run_pending()
    time.sleep(60)
```

## 5. 与标准方案对比

| 维度 | 生物启发式 | 标准向量记忆 |
|------|------------|--------------|
| **记忆保留** | 更自然 (遗忘曲线) | 永久存储 |
